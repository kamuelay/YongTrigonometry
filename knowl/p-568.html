\begin{mybox}[Properties of Composition of Trigonometric Functions]

\begin{equation*} \sin(\sin^{-1}x)=x\mbox{ when } -1\leq x\leq1 \end{equation*}
\begin{equation*} \cos(\cos^{-1}x)=x\mbox{ when } -1\leq x\leq1 \end{equation*}
\begin{equation*} \tan(\tan^{-1}x)=x\mbox{ when } -\infty\leq x\leq\infty \end{equation*}
\begin{equation*} \sin^{-1}(\sin x)=x\mathbf{{only} when }-\frac{\pi}{2}\leq x\leq\frac{\pi}{2} \end{equation*}
