\begin{equation*} \sin\theta=\frac{\mbox{opposite} }{\mbox{hypotenuse} }=\frac{3}{5} \end{equation*}

and

\begin{equation*} \tan\theta=\frac{\mbox{opposite} }{\mbox{adjacent} }=\frac{3}{-4}=-\frac{3}{4} \end{equation*}
