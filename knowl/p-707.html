\begin{mybox}[Product-to-Sum Formulas]

\begin{equation*} \sin\alpha\cos\beta=\frac{1}{2}[\sin(\alpha+\beta)+\sin(\alpha-\beta)] \end{equation*}
\begin{equation*} \cos\alpha\sin\beta=\frac{1}{2}[\sin(\alpha+\beta)-\sin(\alpha-\beta)] \end{equation*}
\begin{equation*} \cos\alpha\cos\beta=\frac{1}{2}[\cos(\alpha+\beta)+\cos(\alpha-\beta)] \end{equation*}
\begin{equation*} \sin\alpha\sin\beta=\frac{1}{2}[\cos(\alpha-\beta)-\cos(\alpha+\beta)] \end{equation*}
