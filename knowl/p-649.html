\begin{mybox}[Double-Angle Formulas]

\begin{align*} \sin 2\theta \amp =\amp 2\sin \theta \cos \theta \end{align*}
\begin{align*} \cos 2\theta \amp =\amp \cos^2 \theta - \sin^2 \theta\\ \amp =\amp 1-2\sin^2 \theta\\ \amp =\amp 2\cos^2 \theta-1 \end{align*}
\begin{align*} \tan 2 \theta \amp =\amp \frac{2\tan \theta}{1-\tan^2 \theta} \end{align*}
